function buildClock(el){
  const f=document.createDocumentFragment();
  // Ziffern 12/3/6/9
  [[12,50,16],[3,84,50],[6,50,84],[9,16,50]].forEach(([n,x,y])=>{
    const d=document.createElement('div');d.className='num';d.textContent=n;
    d.style.left=x+'%';d.style.top=y+'%';d.style.transform='translate(-50%,-50%)';f.appendChild(d);
  });
  // Hände (auf 23:55:00 = „5 vor 12")
  ['hour','minute','second'].forEach((c,i)=>{
    const h=document.createElement('div');h.className='hand '+c;
    h.style.transform='rotate('+[330,330,0][i]+'deg)';f.appendChild(h);
  });
  const cap=document.createElement('div');cap.className='cap';
  const cr=document.createElement('div');cr.className='crown';
  f.appendChild(cap);f.appendChild(cr);el.appendChild(f);
}

Urgency-Engine — Glow nach Zeitdruck

Du wolltest: ruhig am Anfang, intensiv kurz vor 12. Das löst eine einzige Variable: --tj-urgency (0 → 1). Jeder Glow im Kit skaliert automatisch mit. Schieb den Regler und beobachte Buttons, Uhr & Fenster.

Interaktiv
Ruhig · 23:42
23:42
noch Zeit
/* CSS: alle Glows leiten sich von --tj-urgency ab */
:root{
  --tj-urgency:0;  /* 0 = ruhig, 1 = 5 vor 12 */
  --tj-glow:calc(8px + var(--tj-urgency) * 26px);
  --tj-glow-strong:calc(14px + var(--tj-urgency) * 44px);
}

// JS: aus der verbleibenden Zeit den Wert setzen
function setUrgency(remainingSec, totalSec){
  const t = 1 - Math.max(0, remainingSec / totalSec);  // 0→1
  document.documentElement.style.setProperty('--tj-urgency', t.toFixed(3));
  // ab 85% Druck in den Alarm-Modus kippen
  document.querySelectorAll('.tj-clock-digital,.tj-clock-analog')
    .forEach(el => el.toggleAttribute('data-critical', t > 0.85));
}
// Beispiel: jede Sekunde aufrufen
setInterval(() => setUrgency(timeLeft, 120), 1000);

// JS — eine Funktion für alles. total & timeLeft in Sekunden.
let total=120, timeLeft=90;
function applyPenalty(timer, seconds=15){
  const oldPct = Math.max(0,timeLeft/total*100);
  timeLeft = Math.max(0, timeLeft - seconds);
  const newPct = timeLeft/total*100;
  // 1) ruckeln + rot färben
  timer.classList.remove('penalty'); void timer.offsetWidth; timer.classList.add('penalty');
  setTimeout(()=>timer.classList.remove('penalty'), 450);
  // 2) "−15s" aufsteigen lassen
  const pop=document.createElement('div'); pop.className='tj-penalty-pop';
  pop.textContent='−'+seconds+'s'; timer.appendChild(pop); setTimeout(()=>pop.remove(),1100);
  // 3) Balken: verlorenes Stück rot aufblitzen, dann zurückspringen
  const fill=timer.querySelector('.fill'), ghost=timer.querySelector('.ghost');
  ghost.style.left=newPct+'%'; ghost.style.width=(oldPct-newPct)+'%'; ghost.style.opacity='.9';
  setTimeout(()=>{ fill.style.width=newPct+'%'; ghost.style.transition='opacity .6s'; ghost.style.opacity='0'; }, 120);
  // Ziffern aktualisieren
  timer.querySelector('.digits').textContent=fmt(timeLeft);
}
function fmt(s){return String(Math.floor(s/60)).padStart(2,'0')+':'+String(s%60).padStart(2,'0');}
document.getElementById('btn').addEventListener('click',()=>applyPenalty(document.getElementById('timer'),15));
